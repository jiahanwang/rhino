466     * Conforms to LC3 specification
467     */
468    static Object coerceTypeImpl(Class<?> type, Object value)
469    {
470        if (value != null && value.getClass() == type) {
471            return value;
472        }
473 
474        switch (getJSTypeCode(value)) {
475 
476        case JSTYPE_NULL:
477            // raise error if type.isPrimitive()
478            if (type.isPrimitive()) {
479                reportConversionError(value, type);
480            }
481            return null;
482 
483        case JSTYPE_UNDEFINED:
484            if (type == ScriptRuntime.StringClass ||
485                type == ScriptRuntime.ObjectClass) {
486                return "undefined";
487            }
488            else {
489                reportConversionError("undefined", type);
490            }
491            break;
492 
493        case JSTYPE_BOOLEAN:
494            // Under LC3, only JS Booleans can be coerced into a Boolean value
495            if (type == Boolean.TYPE ||
496                type == ScriptRuntime.BooleanClass ||
497                type == ScriptRuntime.ObjectClass) {
498                return value;
499            }
500            else if (type == ScriptRuntime.StringClass) {
501                return value.toString();
502            }
503            else {
504                reportConversionError(value, type);
505            }
506            break;
507 
508        case JSTYPE_NUMBER:
509            if (type == ScriptRuntime.StringClass) {
510                return ScriptRuntime.toString(value);
511            }
512            else if (type == ScriptRuntime.ObjectClass) {
513                return coerceToNumber(Double.TYPE, value);
514            }
515            else if ((type.isPrimitive() && type != Boolean.TYPE) ||
516                     ScriptRuntime.NumberClass.isAssignableFrom(type)) {
517                return coerceToNumber(type, value);
518            }
519            else {
520                reportConversionError(value, type);
521            }
522            break;
523 
524        case JSTYPE_STRING:
525            if (type == ScriptRuntime.StringClass || type.isInstance(value)) {
526                return value.toString();
527            }
528            else if (type == Character.TYPE
529                     || type == ScriptRuntime.CharacterClass)
530            {
531                // Special case for converting a single char string to a
532                // character
533                // Placed here because it applies *only* to JS strings,
534                // not other JS objects converted to strings
535                if (((CharSequence)value).length() == 1) {
536                    return Character.valueOf(((CharSequence)value).charAt(0));
537                }
538                else {
539                    return coerceToNumber(type, value);
540                }
541            }
542            else if ((type.isPrimitive() && type != Boolean.TYPE)
543                     || ScriptRuntime.NumberClass.isAssignableFrom(type))
544            {
545                return coerceToNumber(type, value);
546            }
547            else {
548                reportConversionError(value, type);
549            }
550            break;
551 
552        case JSTYPE_JAVA_CLASS:
553            if (value instanceof Wrapper) {
554                value = ((Wrapper)value).unwrap();
555            }
556 
557            if (type == ScriptRuntime.ClassClass ||
558                type == ScriptRuntime.ObjectClass) {
559                return value;
560            }
561            else if (type == ScriptRuntime.StringClass) {
562                return value.toString();
563            }
564            else {
565                reportConversionError(value, type);
566            }
567            break;
568 
569        case JSTYPE_JAVA_OBJECT:
570        case JSTYPE_JAVA_ARRAY:
571            if (value instanceof Wrapper) {
572              value = ((Wrapper)value).unwrap();
573            }
574            if (type.isPrimitive()) {
575                if (type == Boolean.TYPE) {
576                    reportConversionError(value, type);
577                }
578                return coerceToNumber(type, value);
579            }
580            else {
581              if (type == ScriptRuntime.StringClass) {
582                    return value.toString();
583                }
584                else {
585                    if (type.isInstance(value)) {
586                        return value;
587                    }
588                    else {
589                        reportConversionError(value, type);
590                    }
591                }
592            }
593            break;
594 
595        case JSTYPE_OBJECT:
596            if (type == ScriptRuntime.StringClass) {
597                return ScriptRuntime.toString(value);
598            }
599            else if (type.isPrimitive()) {
600                if (type == Boolean.TYPE) {
601                    reportConversionError(value, type);
602                }
603                return coerceToNumber(type, value);
604            }
605            else if (type.isInstance(value)) {
606                return value;
607            }
608            else if (type == ScriptRuntime.DateClass
609                     && value instanceof NativeDate)
610            {
611                double time = ((NativeDate)value).getJSTimeValue();
612                // XXX: This will replace NaN by 0
613                return new Date((long)time);
614            }
615            else if (type.isArray() && value instanceof NativeArray) {
616                // Make a new java array, and coerce the JS array components
617                // to the target (component) type.
618                NativeArray array = (NativeArray) value;
619                long length = array.getLength();
620                Class<?> arrayType = type.getComponentType();
621                Object Result = Array.newInstance(arrayType, (int)length);
622                for (int i = 0 ; i < length ; ++i) {
623                    try  {
624                        Array.set(Result, i, coerceTypeImpl(
625                                arrayType, array.get(i, array)));
626                    }
627                    catch (EvaluatorException ee) {
628                        reportConversionError(value, type);
629                    }
630                }
631 
632                return Result;
633            }
634            else if (value instanceof Wrapper) {
635                value = ((Wrapper)value).unwrap();
636                if (type.isInstance(value))
637                    return value;
638                reportConversionError(value, type);
639            }
640            else if (type.isInterface() && (value instanceof NativeObject
641                    || value instanceof NativeFunction)) {
642                // Try to use function/object as implementation of Java interface.
643                return createInterfaceAdapter(type, (ScriptableObject) value);
644            } else {
645                reportConversionError(value, type);
646            }
647            break;
648        }
649 
650        return value;
651    }

653    protected static Object createInterfaceAdapter(Class<?>type, ScriptableObject so) {
654        // XXX: Currently only instances of ScriptableObject are
655        // supported since the resulting interface proxies should
656        // be reused next time conversion is made and generic
657        // Callable has no storage for it. Weak references can
658        // address it but for now use this restriction.
659 
660        Object key = Kit.makeHashKeyFromPair(COERCED_INTERFACE_KEY, type);
661        Object old = so.getAssociatedValue(key);
662        if (old != null) {
663            // Function was already wrapped
664            return old;
665        }
666        Context cx = Context.getContext();
667        Object glue = InterfaceAdapter.create(cx, type, so);
668        // Store for later retrieval
669        glue = so.associateValue(key, glue);
670        return glue;
671    }

673    private static Object coerceToNumber(Class<?> type, Object value)
674    {
675        Class<?> valueClass = value.getClass();
676 
677        // Character
678        if (type == Character.TYPE || type == ScriptRuntime.CharacterClass) {
679            if (valueClass == ScriptRuntime.CharacterClass) {
680                return value;
681            }
682            return Character.valueOf((char)toInteger(value,
683                                                 ScriptRuntime.CharacterClass,
684                                                 Character.MIN_VALUE,
685                                                 Character.MAX_VALUE));
686        }
687 
688        // Double, Float
689        if (type == ScriptRuntime.ObjectClass ||
690            type == ScriptRuntime.DoubleClass || type == Double.TYPE) {
691            return valueClass == ScriptRuntime.DoubleClass
692                ? value
693                : new Double(toDouble(value));
694        }
695 
696        if (type == ScriptRuntime.FloatClass || type == Float.TYPE) {
697            if (valueClass == ScriptRuntime.FloatClass) {
698                return value;
699            }
700            else {
701                double number = toDouble(value);
702                if (Double.isInfinite(number) || Double.isNaN(number)
703                    || number == 0.0) {
704                    return new Float((float)number);
705                }
706                else {
707                    double absNumber = Math.abs(number);
708                    if (absNumber < Float.MIN_VALUE) {
709                        return new Float((number > 0.0) ? +0.0 : -0.0);
710                    }
711                    else if (absNumber > Float.MAX_VALUE) {
712                        return new Float((number > 0.0) ?
713                                         Float.POSITIVE_INFINITY :
714                                         Float.NEGATIVE_INFINITY);
715                    }
716                    else {
717                        return new Float((float)number);
718                    }
719                }
720            }
721        }
722 
723        // Integer, Long, Short, Byte
724        if (type == ScriptRuntime.IntegerClass || type == Integer.TYPE) {
725            if (valueClass == ScriptRuntime.IntegerClass) {
726                return value;
727            }
728            else {
729                return Integer.valueOf((int)toInteger(value,
730                                                  ScriptRuntime.IntegerClass,
731                                                  Integer.MIN_VALUE,
732                                                  Integer.MAX_VALUE));
733            }
734        }
735 
736        if (type == ScriptRuntime.LongClass || type == Long.TYPE) {
737            if (valueClass == ScriptRuntime.LongClass) {
738                return value;
739            } else {
740                /* Long values cannot be expressed exactly in doubles.
741                 * We thus use the largest and smallest double value that
742                 * has a value expressible as a long value. We build these
743                 * numerical values from their hexidecimal representations
744                 * to avoid any problems caused by attempting to parse a
745                 * decimal representation.
746                 */
747                final double max = Double.longBitsToDouble(0x43dfffffffffffffL);
748                final double min = Double.longBitsToDouble(0xc3e0000000000000L);
749                return Long.valueOf(toInteger(value,
750                                          ScriptRuntime.LongClass,
751                                          min,
752                                          max));
753            }
754        }
755 
756        if (type == ScriptRuntime.ShortClass || type == Short.TYPE) {
757            if (valueClass == ScriptRuntime.ShortClass) {
758                return value;
759            }
760            else {
761                return Short.valueOf((short)toInteger(value,
762                                                  ScriptRuntime.ShortClass,
763                                                  Short.MIN_VALUE,
764                                                  Short.MAX_VALUE));
765            }
766        }
767 
768        if (type == ScriptRuntime.ByteClass || type == Byte.TYPE) {
769            if (valueClass == ScriptRuntime.ByteClass) {
770                return value;
771            }
772            else {
773                return Byte.valueOf((byte)toInteger(value,
774                                                ScriptRuntime.ByteClass,
775                                                Byte.MIN_VALUE,
776                                                Byte.MAX_VALUE));
777            }
778        }
779 
780        return new Double(toDouble(value));
781    }

784    private static double toDouble(Object value)
785    {
786        if (value instanceof Number) {
787            return ((Number)value).doubleValue();
788        }
789        else if (value instanceof String) {
790            return ScriptRuntime.toNumber((String)value);
791        }
792        else if (value instanceof Scriptable) {
793            if (value instanceof Wrapper) {
794                // XXX: optimize tail-recursion?
795                return toDouble(((Wrapper)value).unwrap());
796            }
797            else {
798                return ScriptRuntime.toNumber(value);
799            }
800        }
801        else {
802            Method meth;
803            try {
804                meth = value.getClass().getMethod("doubleValue",
805                                                          (Class [])null);
806            }
807            catch (NoSuchMethodException e) {
808                meth = null;
809            }
810            catch (SecurityException e) {
811                meth = null;
812            }
813            if (meth != null) {
814                try {
815                    return ((Number)meth.invoke(value,
816                                                        (Object [])null)).doubleValue();
817                }
818                catch (IllegalAccessException e) {
819                    // XXX: ignore, or error message?
820                    reportConversionError(value, Double.TYPE);
821                }
822                catch (InvocationTargetException e) {
823                    // XXX: ignore, or error message?
824                    reportConversionError(value, Double.TYPE);
825                }
826            }
827            return ScriptRuntime.toNumber(value.toString());
828        }

849            // Convert to string first, for more readable message
850            reportConversionError(ScriptRuntime.toString(value), type);
851        }
852        return (long)d;
