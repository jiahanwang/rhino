445                }
446            }
447 
448            if (member.isMethod()) {
449                result = member.invoke(thisObj, invokeArgs);
450                checkMethodResult = true;
451            } else {
452                result = member.newInstance(invokeArgs);
453            }
454 
455        }
456 
457        if (checkMethodResult) {
458            if (hasVoidReturn) {
459                result = Undefined.instance;
460            } else if (returnTypeTag == JAVA_UNSUPPORTED_TYPE) {
461                result = cx.getWrapFactory().wrap(cx, scope, result, null);
462            }
463            // XXX: the code assumes that if returnTypeTag == JAVA_OBJECT_TYPE
464            // then the Java method did a proper job of converting the
465            // result to JS primitive or Scriptable to avoid
466            // potentially costly Context.javaToJS call.
467        }
468 
469        return result;
470    }

472    /**
473     * Return new {@link Scriptable} instance using the default
474     * constructor for the class of the underlying Java method.
475     * Return null to indicate that the call method should be used to create
476     * new objects.
477     */
478    @Override
479    public Scriptable createObject(Context cx, Scriptable scope) {
480        if (member.isCtor() || parmsLength == VARARGS_CTOR) {
481            return null;
482        }
483        Scriptable result;
484        try {
485            result = (Scriptable) member.getDeclaringClass().newInstance();
486        } catch (Exception ex) {
487            throw Context.throwAsScriptRuntimeEx(ex);
488        }
489 
490        result.setPrototype(getClassPrototype());
491        result.setParentScope(getParentScope());
492        return result;
493    }

496        return parmsLength == VARARGS_METHOD;
497    }

503    private void readObject(ObjectInputStream in)
504        throws IOException, ClassNotFoundException
505    {
506        in.defaultReadObject();
507        if (parmsLength > 0) {
508            Class<?>[] types = member.argTypes;
509            typeTags = new byte[parmsLength];
510            for (int i = 0; i != parmsLength; ++i) {
511                typeTags[i] = (byte)getTypeTag(types[i]);
512            }
513        }
514        if (member.isMethod()) {
515            Method method = member.method();
516            Class<?> returnType = method.getReturnType();
517            if (returnType == Void.TYPE) {
518                hasVoidReturn = true;
519            } else {
520                returnTypeTag = getTypeTag(returnType);
521            }
522        }
