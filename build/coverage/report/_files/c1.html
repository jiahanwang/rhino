693    public void defineOwnProperty(Context cx, Object key, ScriptableObject desc) {
694      if (key instanceof String) {
695        String name = (String) key;
696        int info = findInstanceIdInfo(name);
697        if (info != 0) {
698            int id = (info & 0xFFFF);
699            if (isAccessorDescriptor(desc)) {
700              delete(id); // it will be replaced with a slot
701            } else {
702              checkPropertyDefinition(desc);
703              ScriptableObject current = getOwnPropertyDescriptor(cx, key);
704              checkPropertyChange(name, current, desc);
705              int attr = (info >>> 16);
706              Object value = getProperty(desc, "value");
707              if (value != NOT_FOUND && (attr & READONLY) == 0) {
708                Object currentValue = getInstanceIdValue(id);
709                if (!sameValue(value, currentValue)) {
710                  setInstanceIdValue(id, value);
711                }
712              }
713              setAttributes(name, applyDescriptorToAttributeBitset(attr, desc));
714              return;
715            }
716        }
717        if (prototypeValues != null) {
718            int id = prototypeValues.findId(name);
719            if (id != 0) {
720              if (isAccessorDescriptor(desc)) {
721                prototypeValues.delete(id); // it will be replaced with a slot
722              } else {
723                checkPropertyDefinition(desc);
724                ScriptableObject current = getOwnPropertyDescriptor(cx, key);
725                checkPropertyChange(name, current, desc);
726                int attr = prototypeValues.getAttributes(id);
727                Object value = getProperty(desc, "value");
728                if (value != NOT_FOUND && (attr & READONLY) == 0) {
729                  Object currentValue = prototypeValues.get(id);
730                  if (!sameValue(value, currentValue)) {
731                    prototypeValues.set(id, this, value);
732                  }
733                }
734                prototypeValues.setAttributes(id, applyDescriptorToAttributeBitset(attr, desc));
735                return;
736              }
737            }
738        }
739      }

741    }
