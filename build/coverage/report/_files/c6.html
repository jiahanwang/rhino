382        for (int i = 0; i < interfacesCount; i++) {
383            Method[] methods = interfaces[i].getMethods();
384            for (int j = 0; j < methods.length; j++) {
385                Method method = methods[j];
386                int mods = method.getModifiers();
387                if (Modifier.isStatic(mods) || Modifier.isFinal(mods)) {
388                    continue;
389                }
390                String methodName = method.getName();
391                Class<?>[] argTypes = method.getParameterTypes();
392                if (!functionNames.has(methodName)) {
393                    try {
394                        superClass.getMethod(methodName, argTypes);
395                        // The class we're extending implements this method and
396                        // the JavaScript object doesn't have an override. See
397                        // bug 61226.
398                        continue;
399                    } catch (NoSuchMethodException e) {
400                        // Not implemented by superclass; fall through
401                    }
402                }
403                // make sure to generate only one instance of a particular
404                // method/signature.
405                String methodSignature = getMethodSignature(method, argTypes);
406                String methodKey = methodName + methodSignature;
407                if (! generatedOverrides.has(methodKey)) {
408                    generateMethod(cfw, adapterName, methodName, argTypes,
409                                   method.getReturnType(), true);
410                    generatedOverrides.put(methodKey, 0);
411                    generatedMethods.put(methodName, 0);
412                }
413            }
414        }
415 
416        // Now, go through the superclass's methods, checking for abstract
417        // methods or additional methods to override.
418 
419        // generate any additional overrides that the object might contain.
420        Method[] methods = getOverridableMethods(superClass);
421        for (int j = 0; j < methods.length; j++) {
422            Method method = methods[j];
423            int mods = method.getModifiers();
424            // if a method is marked abstract, must implement it or the
425            // resulting class won't be instantiable. otherwise, if the object
426            // has a property of the same name, then an override is intended.
427            boolean isAbstractMethod = Modifier.isAbstract(mods);
428            String methodName = method.getName();
429            if (isAbstractMethod || functionNames.has(methodName)) {
430                // make sure to generate only one instance of a particular
431                // method/signature.
432                Class<?>[] argTypes = method.getParameterTypes();
433                String methodSignature = getMethodSignature(method, argTypes);
434                String methodKey = methodName + methodSignature;
435                if (! generatedOverrides.has(methodKey)) {
436                    generateMethod(cfw, adapterName, methodName, argTypes,
437                                   method.getReturnType(), true);
438                    generatedOverrides.put(methodKey, 0);
439                    generatedMethods.put(methodName, 0);
440 
441                    // if a method was overridden, generate a "super$method"
442                    // which lets the delegate call the superclass' version.
443                    if (!isAbstractMethod) {
444                        generateSuper(cfw, adapterName, superName,
445                                      methodName, methodSignature,
446                                      argTypes, method.getReturnType());
447                    }
448                }
449            }
450        }
451 
452        // Generate Java methods for remaining properties that are not
453        // overrides.
454        ObjToIntMap.Iterator iter = new ObjToIntMap.Iterator(functionNames);
455        for (iter.start(); !iter.done(); iter.next()) {
456            String functionName = (String)iter.getKey();
457            if (generatedMethods.has(functionName))
458                continue;
459            int length = iter.getValue();
460            Class<?>[] parms = new Class[length];
461            for (int k=0; k < length; k++)
462                parms[k] = ScriptRuntime.ObjectClass;
463            generateMethod(cfw, adapterName, functionName, parms,
464                           ScriptRuntime.ObjectClass, false);
465        }
466        return cfw.toByteArray();
467    }

469    static Method[] getOverridableMethods(Class<?> clazz)
470    {
471        ArrayList<Method> list = new ArrayList<Method>();
472        HashSet<String> skip = new HashSet<String>();
473        // Check superclasses before interfaces so we always choose
474        // implemented methods over abstract ones, even if a subclass
475        // re-implements an interface already implemented in a superclass
476        // (e.g. java.util.ArrayList)
477        for (Class<?> c = clazz; c != null; c = c.getSuperclass()) {
478            appendOverridableMethods(c, list, skip);
479        }
480        for (Class<?> c = clazz; c != null; c = c.getSuperclass()) {
481            for (Class<?> intf: c.getInterfaces())
482                appendOverridableMethods(intf, list, skip);
483        }
484        return list.toArray(new Method[list.size()]);
485    }
486 
487    private static void appendOverridableMethods(Class<?> c,
488            ArrayList<Method> list, HashSet<String> skip)
489    {
490        Method[] methods = c.getDeclaredMethods();
491        for (int i = 0; i < methods.length; i++) {
492            String methodKey = methods[i].getName() +
493                getMethodSignature(methods[i],
494                        methods[i].getParameterTypes());
495            if (skip.contains(methodKey))
496                continue; // skip this method
497            int mods = methods[i].getModifiers();
498            if (Modifier.isStatic(mods))
499                continue;
500            if (Modifier.isFinal(mods)) {
501                // Make sure we don't add a final method to the list
502                // of overridable methods.
503                skip.add(methodKey);
504                continue;
505            }
506            if (Modifier.isPublic(mods) || Modifier.isProtected(mods)) {
507                list.add(methods[i]);
508                skip.add(methodKey);
509            }
510        }
511    }

513    static Class<?> loadAdapterClass(String className, byte[] classBytes)
514    {
515        Object staticDomain;
516        Class<?> domainClass = SecurityController.getStaticSecurityDomainClass();
517        if(domainClass == CodeSource.class || domainClass == ProtectionDomain.class) {

519            ProtectionDomain protectionDomain = SecurityUtilities.getScriptProtectionDomain();

521                protectionDomain = JavaAdapter.class.getProtectionDomain();

540    public static Function getFunction(Scriptable obj, String functionName)
541    {
542        Object x = ScriptableObject.getProperty(obj, functionName);
543        if (x == Scriptable.NOT_FOUND) {
544            // This method used to swallow the exception from calling

557    /**
558     * Utility method which dynamically binds a Context to the current thread,
559     * if none already exists.
560     */
561    public static Object callMethod(ContextFactory factory,
562                                    final Scriptable thisObj,
563                                    final Function f, final Object[] args,
564                                    final long argsToWrap)
565    {
566        if (f == null) {
567            // See comments in getFunction
568            return null;
569        }
570        if (factory == null) {
571            factory = ContextFactory.getGlobal();
572        }
573 
574        final Scriptable scope = f.getParentScope();
575        if (argsToWrap == 0) {
576            return Context.call(factory, f, scope, thisObj, args);
577        }
578 
579        Context cx = Context.getCurrentContext();
580        if (cx != null) {
581            return doCall(cx, scope, thisObj, f, args, argsToWrap);
582        } else {
583            return factory.call(new ContextAction() {
584                public Object run(Context cx)
585                {
586                    return doCall(cx, scope, thisObj, f, args, argsToWrap);
587                }
588            });
589        }
590    }

592    private static Object doCall(Context cx, Scriptable scope,
593                                 Scriptable thisObj, Function f,
594                                 Object[] args, long argsToWrap)
595    {
596        // Wrap the rest of objects
597        for (int i = 0; i != args.length; ++i) {
598            if (0 != (argsToWrap & (1 << i))) {
599                Object arg = args[i];
600                if (!(arg instanceof Scriptable)) {
601                    args[i] = cx.getWrapFactory().wrap(cx, scope, arg,
602                                                       null);
603                }
604            }
605        }
606        return f.call(cx, scope, thisObj, args);
607    }

622    private static void generateCtor(ClassFileWriter cfw, String adapterName,
623                                     String superName, Constructor<?> superCtor)
624    {
625        short locals = 3; // this + factory + delegee
626        Class<?>[] parameters = superCtor.getParameterTypes();
627 
628        // Note that we swapped arguments in app-facing constructors to avoid
629        // conflicting signatures with serial constructor defined below.
630        if (parameters.length == 0) {
631            cfw.startMethod("<init>",
632                        "(Lorg/mozilla/javascript/Scriptable;"
633                        +"Lorg/mozilla/javascript/ContextFactory;)V",
634                        ClassFileWriter.ACC_PUBLIC);
635 
636            // Invoke base class constructor
637            cfw.add(ByteCode.ALOAD_0);  // this
638            cfw.addInvoke(ByteCode.INVOKESPECIAL, superName, "<init>", "()V");
639        } else {
640            StringBuilder sig = new StringBuilder(
641                    "(Lorg/mozilla/javascript/Scriptable;"
642                    +"Lorg/mozilla/javascript/ContextFactory;");
643            int marker = sig.length(); // lets us reuse buffer for super signature
644            for (Class<?> c : parameters) {
645                appendTypeString(sig, c);
646            }
647            sig.append(")V");
648            cfw.startMethod("<init>", sig.toString(), ClassFileWriter.ACC_PUBLIC);
649 
650            // Invoke base class constructor
651            cfw.add(ByteCode.ALOAD_0);  // this
652            short paramOffset = 3;
653            for (Class<?> parameter : parameters) {
654                paramOffset += generatePushParam(cfw, paramOffset, parameter);
655            }
656            locals = paramOffset;
657            sig.delete(1, marker);
658            cfw.addInvoke(ByteCode.INVOKESPECIAL, superName, "<init>", sig.toString());
659        }
660 
661        // Save parameter in instance variable "delegee"
662        cfw.add(ByteCode.ALOAD_0);  // this
663        cfw.add(ByteCode.ALOAD_1);  // first arg: Scriptable delegee
664        cfw.add(ByteCode.PUTFIELD, adapterName, "delegee",
665                "Lorg/mozilla/javascript/Scriptable;");
666 
667        // Save parameter in instance variable "factory"
668        cfw.add(ByteCode.ALOAD_0);  // this
669        cfw.add(ByteCode.ALOAD_2);  // second arg: ContextFactory instance
670        cfw.add(ByteCode.PUTFIELD, adapterName, "factory",
671                "Lorg/mozilla/javascript/ContextFactory;");
672 
673        cfw.add(ByteCode.ALOAD_0);  // this for the following PUTFIELD for self
674        // create a wrapper object to be used as "this" in method calls
675        cfw.add(ByteCode.ALOAD_1);  // the Scriptable delegee
676        cfw.add(ByteCode.ALOAD_0);  // this
677        cfw.addInvoke(ByteCode.INVOKESTATIC,
678                      "org/mozilla/javascript/JavaAdapter",
679                      "createAdapterWrapper",
680                      "(Lorg/mozilla/javascript/Scriptable;"
681                      +"Ljava/lang/Object;"
682                      +")Lorg/mozilla/javascript/Scriptable;");
683        cfw.add(ByteCode.PUTFIELD, adapterName, "self",
684                "Lorg/mozilla/javascript/Scriptable;");
685 
686        cfw.add(ByteCode.RETURN);
687        cfw.stopMethod(locals);
688    }

779    /**
780     * Generates code to wrap Java arguments into Object[].
781     * Non-primitive Java types are left as-is pending conversion
782     * in the helper method. Leaves the array object on the top of the stack.
783     */
784    static void generatePushWrappedArgs(ClassFileWriter cfw,
785                                        Class<?>[] argTypes,
786                                        int arrayLength)
787    {
788        // push arguments
789        cfw.addPush(arrayLength);
790        cfw.add(ByteCode.ANEWARRAY, "java/lang/Object");
791        int paramOffset = 1;
792        for (int i = 0; i != argTypes.length; ++i) {
793            cfw.add(ByteCode.DUP); // duplicate array reference
794            cfw.addPush(i);
795            paramOffset += generateWrapArg(cfw, paramOffset, argTypes[i]);
796            cfw.add(ByteCode.AASTORE);
797        }
798    }
799 
800    /**
801     * Generates code to wrap Java argument into Object.
802     * Non-primitive Java types are left unconverted pending conversion
803     * in the helper method. Leaves the wrapper object on the top of the stack.
804     */
805    private static int generateWrapArg(ClassFileWriter cfw, int paramOffset,
806                                       Class<?> argType)
807    {
808        int size = 1;
809        if (!argType.isPrimitive()) {
810            cfw.add(ByteCode.ALOAD, paramOffset);
811 
812        } else if (argType == Boolean.TYPE) {
813            // wrap boolean values with java.lang.Boolean.
814            cfw.add(ByteCode.NEW, "java/lang/Boolean");
815            cfw.add(ByteCode.DUP);
816            cfw.add(ByteCode.ILOAD, paramOffset);
817            cfw.addInvoke(ByteCode.INVOKESPECIAL, "java/lang/Boolean",
818                          "<init>", "(Z)V");
819 
820        } else if (argType == Character.TYPE) {
821            // Create a string of length 1 using the character parameter.
822            cfw.add(ByteCode.ILOAD, paramOffset);
823            cfw.addInvoke(ByteCode.INVOKESTATIC, "java/lang/String",
824                          "valueOf", "(C)Ljava/lang/String;");
825 
826        } else {
827            // convert all numeric values to java.lang.Double.
828            cfw.add(ByteCode.NEW, "java/lang/Double");
829            cfw.add(ByteCode.DUP);
830            String typeName = argType.getName();
831            switch (typeName.charAt(0)) {
832            case 'b':
833            case 's':
834            case 'i':
835                // load an int value, convert to double.
836                cfw.add(ByteCode.ILOAD, paramOffset);
837                cfw.add(ByteCode.I2D);
838                break;
839            case 'l':
840                // load a long, convert to double.
841                cfw.add(ByteCode.LLOAD, paramOffset);
842                cfw.add(ByteCode.L2D);
843                size = 2;
844                break;
845            case 'f':
846                // load a float, convert to double.
847                cfw.add(ByteCode.FLOAD, paramOffset);
