481    /**
482     * Determine which of two signatures is the closer fit.
483     * Returns one of PREFERENCE_EQUAL, PREFERENCE_FIRST_ARG,
484     * PREFERENCE_SECOND_ARG, or PREFERENCE_AMBIGUOUS.
485     */
486    private static int preferSignature(Object[] args,
487                                       Class<?>[] sig1,
488                                       boolean vararg1,
489                                       Class<?>[] sig2,
490                                       boolean vararg2 )
491    {
492        int totalPreference = 0;
493        for (int j = 0; j < args.length; j++) {
494            Class<?> type1 = vararg1 && j >= sig1.length ? sig1[sig1.length-1] : sig1[j];
495            Class<?> type2 = vararg2 && j >= sig2.length ? sig2[sig2.length-1] : sig2[j];
496            if (type1 == type2) {
497                continue;
498            }
499            Object arg = args[j];
500 
501            // Determine which of type1, type2 is easier to convert from arg.
502 
503            int rank1 = NativeJavaObject.getConversionWeight(arg, type1);
504            int rank2 = NativeJavaObject.getConversionWeight(arg, type2);
505 
506            int preference;
507            if (rank1 < rank2) {
508                preference = PREFERENCE_FIRST_ARG;
509            } else if (rank1 > rank2) {
510                preference = PREFERENCE_SECOND_ARG;
511            } else {
512                // Equal ranks
513                if (rank1 == NativeJavaObject.CONVERSION_NONTRIVIAL) {
514                    if (type1.isAssignableFrom(type2)) {
515                        preference = PREFERENCE_SECOND_ARG;
516                    } else if (type2.isAssignableFrom(type1)) {
517                        preference = PREFERENCE_FIRST_ARG;
518                    } else {
519                        preference = PREFERENCE_AMBIGUOUS;
520                    }
521                } else {
522                    preference = PREFERENCE_AMBIGUOUS;
523                }
524            }
525 
526            totalPreference |= preference;
527 
528            if (totalPreference == PREFERENCE_AMBIGUOUS) {
529                break;
530            }
531        }
532        return totalPreference;
533    }
