271    // Returns the next token without consuming it.
272    // If previous token was consumed, calls scanner to get new token.
273    // If previous token was -not- consumed, returns it (idempotent).
274    //
275    // This function will not return a newline (Token.EOL - instead, it
276    // gobbles newlines until it finds a non-newline token, and flags
277    // that token as appearing just after a newline.
278    //
279    // This function will also not return a Token.COMMENT.  Instead, it
280    // records comments in the scannedComments list.  If the token
281    // returned by this function immediately follows a jsdoc comment,
282    // the token is flagged as such.
283    //
284    // Note that this function always returned the un-flagged token!
285    // The flags, if any, are saved in currentFlaggedToken.
286    private int peekToken()
287        throws IOException
288    {
289        // By far the most common case:  last token hasn't been consumed,
290        // so return already-peeked token.
291        if (currentFlaggedToken != Token.EOF) {
292            return currentToken;
293        }
294 
295        int lineno = ts.getLineno();
296        int tt = ts.getToken();
297        boolean sawEOL = false;
298 
299        // process comments and whitespace
300        while (tt == Token.EOL || tt == Token.COMMENT) {
301            if (tt == Token.EOL) {
302                lineno++;
303                sawEOL = true;
304            } else {
305                if (compilerEnv.isRecordingComments()) {
306                    String comment = ts.getAndResetCurrentComment();
307                    recordComment(lineno, comment);
308                    // Comments may contain multiple lines, get the number
309                    // of EoLs and increase the lineno
310                    lineno += getNumberOfEols(comment);
311                }
312            }
313            tt = ts.getToken();
314        }
315 
316        currentToken = tt;
317        currentFlaggedToken = tt | (sawEOL ? TI_AFTER_EOL : 0);
318        return currentToken;  // return unflagged token
319    }

508    private AstRoot parse() throws IOException
509    {
510        int pos = 0;
511        AstRoot root = new AstRoot(pos);
512        currentScope = currentScriptOrFn = root;
513 
514        int baseLineno = ts.lineno;  // line number where source starts
515        int end = pos;  // in case source is empty
516 
517        boolean inDirectivePrologue = true;
518        boolean savedStrictMode = inUseStrictDirective;
519        // TODO: eval code should get strict mode from invoking code
520        inUseStrictDirective = false;
521 
522        try {
523            for (;;) {
524                int tt = peekToken();
525                if (tt <= Token.EOF) {
526                    break;
527                }
528 
529                AstNode n;
530                if (tt == Token.FUNCTION) {
531                    consumeToken();
532                    try {
533                        n = function(calledByCompileFunction
534                                     ? FunctionNode.FUNCTION_EXPRESSION
535                                     : FunctionNode.FUNCTION_STATEMENT);
536                    } catch (ParserException e) {
537                        break;
538                    }
539                } else {
540                    n = statement();
541                    if (inDirectivePrologue) {
542                        String directive = getDirective(n);
543                        if (directive == null) {
544                            inDirectivePrologue = false;
545                        } else if (directive.equals("use strict")) {
546                            inUseStrictDirective = true;
547                            root.setInStrictMode(true);
548                        }
549                    }
550 
551                }
552                end = getNodeEnd(n);
553                root.addChildToBack(n);
554                n.setParent(root);
555            }
556        } catch (StackOverflowError ex) {
557            String msg = lookupMessage("msg.too.deep.parser.recursion");
558            if (!compilerEnv.isIdeMode())
559                throw Context.reportRuntimeError(msg, sourceURI,
560                                                 ts.lineno, null, 0);
561        } finally {
562            inUseStrictDirective = savedStrictMode;
563        }
564 
565        if (this.syntaxErrorCount != 0) {
566            String msg = String.valueOf(this.syntaxErrorCount);
567            msg = lookupMessage("msg.got.syntax.errors", msg);
568            if (!compilerEnv.isIdeMode())
569                throw errorReporter.runtimeError(msg, sourceURI, baseLineno,
570                                                 null, 0);
571        }
572 
573        // add comments to root in lexical order
574        if (scannedComments != null) {
575            // If we find a comment beyond end of our last statement or
576            // function, extend the root bounds to the end of that comment.
577            int last = scannedComments.size() - 1;
578            end = Math.max(end, getNodeEnd(scannedComments.get(last)));
579            for (Comment c : scannedComments) {
580                root.addComment(c);
581            }
582        }
583 
584        root.setLength(end - pos);
585        root.setSourceName(sourceURI);
586        root.setBaseLineno(baseLineno);
587        root.setEndLineno(ts.lineno);
588        return root;
589    }

591    private AstNode parseFunctionBody()
592        throws IOException
593    {
594        boolean isExpressionClosure = false;
595        if (!matchToken(Token.LC)) {
596            if (compilerEnv.getLanguageVersion() < Context.VERSION_1_8) {
597                reportError("msg.no.brace.body");
598            } else {
599                isExpressionClosure = true;
600            }
601        }
602        ++nestingOfFunction;
603        int pos = ts.tokenBeg;
604        Block pn = new Block(pos);  // starts at LC position
605 
606        boolean inDirectivePrologue = true;
607        boolean savedStrictMode = inUseStrictDirective;
608        // Don't set 'inUseStrictDirective' to false: inherit strict mode.
609 
610        pn.setLineno(ts.lineno);
611        try {
612            if (isExpressionClosure) {
613                ReturnStatement n = new ReturnStatement(ts.lineno);
614                n.setReturnValue(assignExpr());
615                // expression closure flag is required on both nodes
616                n.putProp(Node.EXPRESSION_CLOSURE_PROP, Boolean.TRUE);
617                pn.putProp(Node.EXPRESSION_CLOSURE_PROP, Boolean.TRUE);
618                pn.addStatement(n);
619            } else {
620                bodyLoop: for (;;) {
621                    AstNode n;
622                    int tt = peekToken();
623                    switch (tt) {
624                        case Token.ERROR:
625                        case Token.EOF:
626                        case Token.RC:
627                            break bodyLoop;
628 
629                        case Token.FUNCTION:
630                            consumeToken();
631                            n = function(FunctionNode.FUNCTION_STATEMENT);
632                            break;
633                        default:
634                            n = statement();
635                            if (inDirectivePrologue) {
636                                String directive = getDirective(n);
637                                if (directive == null) {
638                                    inDirectivePrologue = false;
639                                } else if (directive.equals("use strict")) {
640                                    inUseStrictDirective = true;
641                                }
642                            }
643                            break;
644                    }
645                    pn.addStatement(n);
646                }
647            }
648        } catch (ParserException e) {
649            // Ignore it
650        } finally {
651            --nestingOfFunction;
652            inUseStrictDirective = savedStrictMode;
653        }
654 
655        int end = ts.tokenEnd;
656        getAndResetJsDoc();
657        if (!isExpressionClosure && mustMatchToken(Token.RC, "msg.no.brace.after.body"))
658            end = ts.tokenEnd;
659        pn.setLength(end - pos);
660        return pn;
661    }
