386    static boolean isGenerator(ScriptNode node)
387    {
388        return (node.getType() == Token.FUNCTION ) &&
389                ((FunctionNode)node).isGenerator();

392    // How dispatch to generators works:
393    // Two methods are generated corresponding to a user-written generator.
394    // One of these creates a generator object (NativeGenerator), which is
395    // returned to the user. The other method contains all of the body code
396    // of the generator.
397    // When a user calls a generator, the call() method dispatches control to
398    // to the method that creates the NativeGenerator object. Subsequently when
399    // the user invokes .next(), .send() or any such method on the generator
400    // object, the resumeGenerator() below dispatches the call to the
401    // method corresponding to the generator body. As a matter of convention
402    // the generator body is given the name of the generator activation function
403    // appended by "_gen".
404    private void generateResumeGenerator(ClassFileWriter cfw)
405    {
406        boolean hasGenerators = false;
407        for (int i=0; i < scriptOrFnNodes.length; i++) {
408            if (isGenerator(scriptOrFnNodes[i]))
409                    hasGenerators = true;
410        }
411 
412        // if there are no generators defined, we don't implement a
413        // resumeGenerator(). The base class provides a default implementation.
414        if (!hasGenerators)
415            return;
416 
417        cfw.startMethod("resumeGenerator",
418                        "(Lorg/mozilla/javascript/Context;" +
419                        "Lorg/mozilla/javascript/Scriptable;" +
420                        "ILjava/lang/Object;" +
421                        "Ljava/lang/Object;)Ljava/lang/Object;",
422                        (short)(ACC_PUBLIC | ACC_FINAL));
423 
424        // load arguments for dispatch to the corresponding *_gen method
425        cfw.addALoad(0);
426        cfw.addALoad(1);
427        cfw.addALoad(2);
428        cfw.addALoad(4);
429        cfw.addALoad(5);
430        cfw.addILoad(3);
431 
432        cfw.addLoadThis();
433        cfw.add(ByteCode.GETFIELD, cfw.getClassName(), ID_FIELD_NAME, "I");
434 
435        int startSwitch = cfw.addTableSwitch(0, scriptOrFnNodes.length - 1);
436        cfw.markTableSwitchDefault(startSwitch);
437        int endlabel = cfw.acquireLabel();
438 
439        for (int i = 0; i < scriptOrFnNodes.length; i++) {
440            ScriptNode n = scriptOrFnNodes[i];
441            cfw.markTableSwitchCase(startSwitch, i, (short)6);
442            if (isGenerator(n)) {
443                String type = "(" +
444                              mainClassSignature +
445                              "Lorg/mozilla/javascript/Context;" +
446                              "Lorg/mozilla/javascript/Scriptable;" +
447                              "Ljava/lang/Object;" +
448                              "Ljava/lang/Object;I)Ljava/lang/Object;";
449                cfw.addInvoke(ByteCode.INVOKESTATIC,
450                              mainClassName,
451                              getBodyMethodName(n) + "_gen",
452                              type);
453                cfw.add(ByteCode.ARETURN);
454            } else {
455                cfw.add(ByteCode.GOTO, endlabel);
456            }
457        }
458 
459        cfw.markLabel(endlabel);
460        pushUndefined(cfw);
461        cfw.add(ByteCode.ARETURN);
462 
463 
464        // this method uses as many locals as there are arguments (hence 6)
465        cfw.stopMethod((short)6);
466    }

476 
477        // Generate code for:
478        // if (!ScriptRuntime.hasTopCall(cx)) {
479        //     return ScriptRuntime.doTopCall(this, cx, scope, thisObj, args);
480        // }
481

483        cfw.addALoad(1); //cx
484        cfw.addInvoke(ByteCode.INVOKESTATIC,
485                      "org/mozilla/javascript/ScriptRuntime",
486                      "hasTopCall",
487                      "(Lorg/mozilla/javascript/Context;"
488                      +")Z");
489        cfw.add(ByteCode.IFNE, nonTopCallLabel);
490        cfw.addALoad(0);
491        cfw.addALoad(1);
492        cfw.addALoad(2);
493        cfw.addALoad(3);
494        cfw.addALoad(4);
495        cfw.addInvoke(ByteCode.INVOKESTATIC,
496                      "org/mozilla/javascript/ScriptRuntime",
497                      "doTopCall",
498                      "(Lorg/mozilla/javascript/Callable;"
499                      +"Lorg/mozilla/javascript/Context;"
500                      +"Lorg/mozilla/javascript/Scriptable;"
501                      +"Lorg/mozilla/javascript/Scriptable;"
502                      +"[Ljava/lang/Object;"
503                      +")Ljava/lang/Object;");
504        cfw.add(ByteCode.ARETURN);
505        cfw.markLabel(nonTopCallLabel);
506 
507        // Now generate switch to call the real methods
508        cfw.addALoad(0);
509        cfw.addALoad(1);
510        cfw.addALoad(2);
511        cfw.addALoad(3);
512        cfw.addALoad(4);
513 
514        int end = scriptOrFnNodes.length;
515        boolean generateSwitch = (2 <= end);
516 
517        int switchStart = 0;
518        int switchStackTop = 0;
519        if (generateSwitch) {
520            cfw.addLoadThis();
521            cfw.add(ByteCode.GETFIELD, cfw.getClassName(), ID_FIELD_NAME, "I");
522            // do switch from (1,  end - 1) mapping 0 to
523            // the default case
524            switchStart = cfw.addTableSwitch(1, end - 1);
525        }
526 
527        for (int i = 0; i != end; ++i) {
528            ScriptNode n = scriptOrFnNodes[i];
529            if (generateSwitch) {
530                if (i == 0) {
531                    cfw.markTableSwitchDefault(switchStart);
532                    switchStackTop = cfw.getStackTop();
533                } else {
534                    cfw.markTableSwitchCase(switchStart, i - 1,
535                                            switchStackTop);
536                }
537            }
538            if (n.getType() == Token.FUNCTION) {
539                OptFunctionNode ofn = OptFunctionNode.get(n);
540                if (ofn.isTargetOfDirectCall()) {
541                    int pcount = ofn.fnode.getParamCount();
542                    if (pcount != 0) {
543                        // loop invariant:
544                        // stack top == arguments array from addALoad4()
545                        for (int p = 0; p != pcount; ++p) {
546                            cfw.add(ByteCode.ARRAYLENGTH);
547                            cfw.addPush(p);
548                            int undefArg = cfw.acquireLabel();
549                            int beyond = cfw.acquireLabel();
550                            cfw.add(ByteCode.IF_ICMPLE, undefArg);
551                            // get array[p]
552                            cfw.addALoad(4);
553                            cfw.addPush(p);
554                            cfw.add(ByteCode.AALOAD);
555                            cfw.add(ByteCode.GOTO, beyond);
556                            cfw.markLabel(undefArg);
557                            pushUndefined(cfw);
558                            cfw.markLabel(beyond);
559                            // Only one push
560                            cfw.adjustStackTop(-1);
561                            cfw.addPush(0.0);
562                            // restore invariant
563                            cfw.addALoad(4);
564                        }
565                    }
566                }
567            }
568            cfw.addInvoke(ByteCode.INVOKESTATIC,
569                          mainClassName,
570                          getBodyMethodName(n),
571                          getBodyMethodSignature(n));
572            cfw.add(ByteCode.ARETURN);
573        }
574        cfw.stopMethod((short)5);
575        // 5: this, cx, scope, js this, args[]
576    }
