450          case Token.RETHROW:
451            updateLineNumber(node);
452            addIndexOp(Token.RETHROW, getLocalBlockRef(node));
453            break;
454 
455          case Token.RETURN:
456            updateLineNumber(node);
457            if (node.getIntProp(Node.GENERATOR_END_PROP, 0) != 0) {
458                // We're in a generator, so change RETURN to GENERATOR_END
459                addIcode(Icode_GENERATOR_END);
460                addUint16(lineNumber & 0xFFFF);
461            } else if (child != null) {
462                visitExpression(child, ECF_TAIL);
463                addToken(Token.RETURN);
464                stackChange(-1);
465            } else {
466                addIcode(Icode_RETUNDEF);
467            }
468            break;
469 
470          case Token.RETURN_RESULT:
471            updateLineNumber(node);
472            addToken(Token.RETURN_RESULT);
473            break;
474 
475          case Token.ENUM_INIT_KEYS:
476          case Token.ENUM_INIT_VALUES:
477          case Token.ENUM_INIT_ARRAY:
478            visitExpression(child, 0);
479            addIndexOp(type, getLocalBlockRef(node));
480            stackChange(-1);
481            break;
482 
483          case Icode_GENERATOR:
484            break;
485 
486          default:
487            throw badTree(node);
488        }
489 
490        if (stackDepth != initialStackDepth) {
491            throw Kit.codeBug();
492        }
493    }

495    private void visitExpression(Node node, int contextFlags)
496    {
497        int type = node.getType();
498        Node child = node.getFirstChild();
499        int savedStackDepth = stackDepth;
500        switch (type) {
501 
502          case Token.FUNCTION:
503            {
504                int fnIndex = node.getExistingIntProp(Node.FUNCTION_PROP);
505                FunctionNode fn = scriptOrFn.getFunctionNode(fnIndex);
506                // See comments in visitStatement for Token.FUNCTION case
507                if (fn.getFunctionType() != FunctionNode.FUNCTION_EXPRESSION) {
508                    throw Kit.codeBug();
509                }
510                addIndexOp(Icode_CLOSURE_EXPR, fnIndex);
511                stackChange(1);
512            }
513            break;
514 
515          case Token.LOCAL_LOAD:
516            {
517                int localIndex = getLocalBlockRef(node);
518                addIndexOp(Token.LOCAL_LOAD, localIndex);
519                stackChange(1);
520            }
521            break;
522 
523          case Token.COMMA:
524            {
525                Node lastChild = node.getLastChild();
526                while (child != lastChild) {
527                    visitExpression(child, 0);
528                    addIcode(Icode_POP);
529                    stackChange(-1);
530                    child = child.getNext();
531                }
532                // Preserve tail context flag if any
533                visitExpression(child, contextFlags & ECF_TAIL);
534            }
535            break;
536 
537          case Token.USE_STACK:
538            // Indicates that stack was modified externally,
539            // like placed catch object
540            stackChange(1);
541            break;
542 
543          case Token.REF_CALL:
544          case Token.CALL:
545          case Token.NEW:
546            {
547                if (type == Token.NEW) {
548                    visitExpression(child, 0);
549                } else {
550                    generateCallFunAndThis(child);
551                }
552                int argCount = 0;
553                while ((child = child.getNext()) != null) {
554                    visitExpression(child, 0);
555                    ++argCount;
556                }
557                int callType = node.getIntProp(Node.SPECIALCALL_PROP,
558                                               Node.NON_SPECIALCALL);
559                if (type != Token.REF_CALL && callType != Node.NON_SPECIALCALL) {
560                    // embed line number and source filename
561                    addIndexOp(Icode_CALLSPECIAL, argCount);
562                    addUint8(callType);
563                    addUint8(type == Token.NEW ? 1 : 0);
564                    addUint16(lineNumber & 0xFFFF);
565                } else {
566                    // Only use the tail call optimization if we're not in a try
567                    // or we're not generating debug info (since the
568                    // optimization will confuse the debugger)
569                    if (type == Token.CALL && (contextFlags & ECF_TAIL) != 0 &&
570                        !compilerEnv.isGenerateDebugInfo() && !itsInTryFlag)
571                    {
572                        type = Icode_TAIL_CALL;
573                    }
574                    addIndexOp(type, argCount);
575                }
576                // adjust stack
577                if (type == Token.NEW) {
578                    // new: f, args -> result
579                    stackChange(-argCount);
580                } else {
581                    // call: f, thisObj, args -> result
582                    // ref_call: f, thisObj, args -> ref
583                    stackChange(-1 - argCount);
584                }
585                if (argCount > itsData.itsMaxCalleeArgs) {
586                    itsData.itsMaxCalleeArgs = argCount;
587                }
588            }
589            break;
590 
591          case Token.AND:
592          case Token.OR:
593            {
594                visitExpression(child, 0);
595                addIcode(Icode_DUP);
596                stackChange(1);
597                int afterSecondJumpStart = iCodeTop;
598                int jump = (type == Token.AND) ? Token.IFNE : Token.IFEQ;
599                addGotoOp(jump);
600                stackChange(-1);
601                addIcode(Icode_POP);
602                stackChange(-1);
603                child = child.getNext();
604                // Preserve tail context flag if any
605                visitExpression(child, contextFlags & ECF_TAIL);
606                resolveForwardGoto(afterSecondJumpStart);
607            }
608            break;
609 
610          case Token.HOOK:
611            {
612                Node ifThen = child.getNext();
613                Node ifElse = ifThen.getNext();
614                visitExpression(child, 0);
615                int elseJumpStart = iCodeTop;
616                addGotoOp(Token.IFNE);
617                stackChange(-1);
618                // Preserve tail context flag if any
619                visitExpression(ifThen, contextFlags & ECF_TAIL);
620                int afterElseJumpStart = iCodeTop;
621                addGotoOp(Token.GOTO);
622                resolveForwardGoto(elseJumpStart);
623                stackDepth = savedStackDepth;
624                // Preserve tail context flag if any
625                visitExpression(ifElse, contextFlags & ECF_TAIL);
626                resolveForwardGoto(afterElseJumpStart);
627            }
628            break;
629 
630          case Token.GETPROP:
631          case Token.GETPROPNOWARN:
632            visitExpression(child, 0);
633            child = child.getNext();
634            addStringOp(type, child.getString());
635            break;
636 
637          case Token.DELPROP:
638            boolean isName = child.getType() == Token.BINDNAME;
639            visitExpression(child, 0);
640            child = child.getNext();
641            visitExpression(child, 0);
642            if (isName) {
643                // special handling for delete name
644                addIcode(Icode_DELNAME);
645            } else {
646                addToken(Token.DELPROP);
647            }
648            stackChange(-1);
649            break;
650 
651          case Token.GETELEM:
652          case Token.BITAND:
653          case Token.BITOR:
654          case Token.BITXOR:
655          case Token.LSH:
656          case Token.RSH:
657          case Token.URSH:
658          case Token.ADD:
659          case Token.SUB:
660          case Token.MOD:
661          case Token.DIV:
662          case Token.MUL:
663          case Token.EQ:
664          case Token.NE:
665          case Token.SHEQ:
666          case Token.SHNE:
667          case Token.IN:
668          case Token.INSTANCEOF:
669          case Token.LE:
670          case Token.LT:
671          case Token.GE:
672          case Token.GT:
673            visitExpression(child, 0);
674            child = child.getNext();
675            visitExpression(child, 0);
676            addToken(type);
677            stackChange(-1);
678            break;
679 
680          case Token.POS:
681          case Token.NEG:
682          case Token.NOT:
683          case Token.BITNOT:
684          case Token.TYPEOF:
685          case Token.VOID:
686            visitExpression(child, 0);
687            if (type == Token.VOID) {
688                addIcode(Icode_POP);
689                addIcode(Icode_UNDEF);
690            } else {
691                addToken(type);
692            }
693            break;
694 
695          case Token.GET_REF:
696          case Token.DEL_REF:
697            visitExpression(child, 0);
698            addToken(type);
699            break;
700 
701          case Token.SETPROP:
702          case Token.SETPROP_OP:
703            {
704                visitExpression(child, 0);
705                child = child.getNext();
706                String property = child.getString();
707                child = child.getNext();
708                if (type == Token.SETPROP_OP) {
709                    addIcode(Icode_DUP);
710                    stackChange(1);
711                    addStringOp(Token.GETPROP, property);
712                    // Compensate for the following USE_STACK
713                    stackChange(-1);
714                }
715                visitExpression(child, 0);
716                addStringOp(Token.SETPROP, property);
717                stackChange(-1);
718            }
719            break;
720 
721          case Token.SETELEM:
722          case Token.SETELEM_OP:
723            visitExpression(child, 0);
724            child = child.getNext();
725            visitExpression(child, 0);
726            child = child.getNext();
727            if (type == Token.SETELEM_OP) {
728                addIcode(Icode_DUP2);
729                stackChange(2);
730                addToken(Token.GETELEM);
731                stackChange(-1);
732                // Compensate for the following USE_STACK
733                stackChange(-1);
734            }
735            visitExpression(child, 0);
736            addToken(Token.SETELEM);
737            stackChange(-2);
738            break;
739 
740          case Token.SET_REF:
741          case Token.SET_REF_OP:
742            visitExpression(child, 0);
743            child = child.getNext();
744            if (type == Token.SET_REF_OP) {
745                addIcode(Icode_DUP);
746                stackChange(1);
747                addToken(Token.GET_REF);
748                // Compensate for the following USE_STACK
749                stackChange(-1);
750            }
751            visitExpression(child, 0);
752            addToken(Token.SET_REF);
753            stackChange(-1);
754            break;
755 
756          case Token.STRICT_SETNAME:
757          case Token.SETNAME:
758            {
759                String name = child.getString();
760                visitExpression(child, 0);
761                child = child.getNext();
762                visitExpression(child, 0);
763                addStringOp(type, name);
764                stackChange(-1);
765            }
766            break;
767 
768          case Token.SETCONST:
769            {
770                String name = child.getString();
771                visitExpression(child, 0);
772                child = child.getNext();
773                visitExpression(child, 0);
774                addStringOp(Icode_SETCONST, name);
775                stackChange(-1);
776            }
777            break;
778 
779          case Token.TYPEOFNAME:
780            {
781                int index = -1;
782                // use typeofname if an activation frame exists
783                // since the vars all exist there instead of in jregs
784                if (itsInFunctionFlag && !itsData.itsNeedsActivation)
785                    index = scriptOrFn.getIndexForNameNode(node);
786                if (index == -1) {
787                    addStringOp(Icode_TYPEOFNAME, node.getString());
788                    stackChange(1);
789                } else {
790                    addVarOp(Token.GETVAR, index);
791                    stackChange(1);
792                    addToken(Token.TYPEOF);
793                }
794            }
795            break;
796 
797          case Token.BINDNAME:
798          case Token.NAME:
799          case Token.STRING:
800            addStringOp(type, node.getString());
801            stackChange(1);
802            break;
803 
804          case Token.INC:
805          case Token.DEC:
806            visitIncDec(node, child);
807            break;
808 
809          case Token.NUMBER:
810            {
811                double num = node.getDouble();
812                int inum = (int)num;
813                if (inum == num) {
814                    if (inum == 0) {
815                        addIcode(Icode_ZERO);
816                        // Check for negative zero
817                        if (1.0 / num < 0.0) {
818                            addToken(Token.NEG);
819                        }
820                    } else if (inum == 1) {
821                        addIcode(Icode_ONE);
822                    } else if ((short)inum == inum) {
823                        addIcode(Icode_SHORTNUMBER);
824                        // write short as uin16 bit pattern
825                        addUint16(inum & 0xFFFF);
826                    } else {
827                        addIcode(Icode_INTNUMBER);
828                        addInt(inum);
829                    }
830                } else {
831                    int index = getDoubleIndex(num);
832                    addIndexOp(Token.NUMBER, index);
833                }
834                stackChange(1);
835            }
836            break;
837 
838          case Token.GETVAR:
839            {
840                if (itsData.itsNeedsActivation) Kit.codeBug();
841                int index = scriptOrFn.getIndexForNameNode(node);
842                addVarOp(Token.GETVAR, index);
843                stackChange(1);
844            }
845            break;
846 
847          case Token.SETVAR:
848            {
849                if (itsData.itsNeedsActivation) Kit.codeBug();
850                int index = scriptOrFn.getIndexForNameNode(child);
851                child = child.getNext();
852                visitExpression(child, 0);
853                addVarOp(Token.SETVAR, index);
854            }
855            break;
856 
857          case Token.SETCONSTVAR:
858            {
859                if (itsData.itsNeedsActivation) Kit.codeBug();
860                int index = scriptOrFn.getIndexForNameNode(child);
861                child = child.getNext();
862                visitExpression(child, 0);
863                addVarOp(Token.SETCONSTVAR, index);
864            }
865            break;
866 
867          case Token.NULL:
868          case Token.THIS:
869          case Token.THISFN:
870          case Token.FALSE:
871          case Token.TRUE:
872            addToken(type);
873            stackChange(1);
874            break;
875 
876          case Token.ENUM_NEXT:
877          case Token.ENUM_ID:
878            addIndexOp(type, getLocalBlockRef(node));
879            stackChange(1);
880            break;
881 
882          case Token.REGEXP:
883            {
884                int index = node.getExistingIntProp(Node.REGEXP_PROP);
885                addIndexOp(Token.REGEXP, index);
886                stackChange(1);
887            }
888            break;
889 
890          case Token.ARRAYLIT:
891          case Token.OBJECTLIT:
892            visitLiteral(node, child);
893            break;
894 
895          case Token.ARRAYCOMP:
896            visitArrayComprehension(node, child, child.getNext());
897            break;
898 
899          case Token.REF_SPECIAL:
900            visitExpression(child, 0);
901            addStringOp(type, (String)node.getProp(Node.NAME_PROP));
902            break;
903 
904          case Token.REF_MEMBER:
905          case Token.REF_NS_MEMBER:
906          case Token.REF_NAME:
907          case Token.REF_NS_NAME:
908            {
909                int memberTypeFlags = node.getIntProp(Node.MEMBER_TYPE_PROP, 0);
910                // generate possible target, possible namespace and member
911                int childCount = 0;
912                do {
913                    visitExpression(child, 0);
914                    ++childCount;
915                    child = child.getNext();
916                } while (child != null);
917                addIndexOp(type, memberTypeFlags);
918                stackChange(1 - childCount);
919            }
920            break;
921 
922          case Token.DOTQUERY:
923            {
924                int queryPC;
925                updateLineNumber(node);
926                visitExpression(child, 0);
927                addIcode(Icode_ENTERDQ);
928                stackChange(-1);
929                queryPC = iCodeTop;
930                visitExpression(child.getNext(), 0);
931                addBackwardGoto(Icode_LEAVEDQ, queryPC);
932            }
933            break;
934 
935          case Token.DEFAULTNAMESPACE :
936          case Token.ESCXMLATTR :
937          case Token.ESCXMLTEXT :
938            visitExpression(child, 0);
939            addToken(type);
940            break;
941 
942          case Token.YIELD:
943            if (child != null) {
944                visitExpression(child, 0);
945            } else {
946                addIcode(Icode_UNDEF);
947                stackChange(1);
948            }
949            addToken(Token.YIELD);
950            addUint16(node.getLineno() & 0xFFFF);
951            break;
952 
953          case Token.WITHEXPR: {
954            Node enterWith = node.getFirstChild();
955            Node with = enterWith.getNext();
956            visitExpression(enterWith.getFirstChild(), 0);
957            addToken(Token.ENTERWITH);
958            stackChange(-1);
959            visitExpression(with.getFirstChild(), 0);
960            addToken(Token.LEAVEWITH);
961            break;
962          }
963 
964          default:
965            throw badTree(node);
966        }
967        if (savedStackDepth + 1 != stackDepth) {
968            Kit.codeBug();
969        }
970    }

972    private void generateCallFunAndThis(Node left)
973    {
974        // Generate code to place on stack function and thisObj
975        int type = left.getType();
976        switch (type) {
977          case Token.NAME: {
978            String name = left.getString();
979            // stack: ... -> ... function thisObj
980            addStringOp(Icode_NAME_AND_THIS, name);
981            stackChange(2);
982            break;
983          }
984          case Token.GETPROP:
985          case Token.GETELEM: {
986            Node target = left.getFirstChild();
987            visitExpression(target, 0);
988            Node id = target.getNext();
989            if (type == Token.GETPROP) {
990                String property = id.getString();
991                // stack: ... target -> ... function thisObj
992                addStringOp(Icode_PROP_AND_THIS, property);
993                stackChange(1);
994            } else {
995                visitExpression(id, 0);
996                // stack: ... target id -> ... function thisObj
997                addIcode(Icode_ELEM_AND_THIS);
998            }
999            break;
1000          }
1001          default:
1002            // Including Token.GETVAR
1003            visitExpression(left, 0);
1004            // stack: ... value -> ... function thisObj
1005            addIcode(Icode_VALUE_AND_THIS);
1006            stackChange(1);
1007            break;
1008        }
1009    }

1012    private void visitIncDec(Node node, Node child)
1013    {
1014        int incrDecrMask = node.getExistingIntProp(Node.INCRDECR_PROP);
1015        int childType = child.getType();
1016        switch (childType) {
1017          case Token.GETVAR : {
1018            if (itsData.itsNeedsActivation) Kit.codeBug();
1019            int i = scriptOrFn.getIndexForNameNode(child);
1020            addVarOp(Icode_VAR_INC_DEC, i);
1021            addUint8(incrDecrMask);
1022            stackChange(1);
1023            break;
1024          }
1025          case Token.NAME : {
1026            String name = child.getString();
1027            addStringOp(Icode_NAME_INC_DEC, name);
1028            addUint8(incrDecrMask);
1029            stackChange(1);
1030            break;
1031          }
1032          case Token.GETPROP : {
1033            Node object = child.getFirstChild();
