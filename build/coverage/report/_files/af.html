507    /**
508     * Get a context associated with the current thread, creating one if need
509     * be. The Context stores the execution state of the JavaScript engine, so
510     * it is required that the context be entered before execution may begin.
511     * Once a thread has entered a Context, then getCurrentContext() may be
512     * called to find the context that is associated with the current thread.
513     * <p>
514     * Calling <code>enterContext()</code> will return either the Context
515     * currently associated with the thread, or will create a new context and
516     * associate it with the current thread. Each call to
517     * <code>enterContext()</code> must have a matching call to
518     * {@link Context#exit()}.
519     * <pre>
520     *      Context cx = contextFactory.enterContext();
521     *      try {
522     *          ...
523     *          cx.evaluateString(...);
524     *      } finally {
525     *          Context.exit();
526     *      }
527     * </pre>
528     * Instead of using <tt>enterContext()</tt>, <tt>exit()</tt> pair consider
529     * using {@link #call(ContextAction)} which guarantees proper association
530     * of Context instances with the current thread.
531     * With this method the above example becomes:
532     * <pre>
533     *      ContextFactory.call(new ContextAction() {
534     *          public Object run(Context cx) {
535     *              ...
536     *              cx.evaluateString(...);
537     *              return null;
538     *          }
539     *      });
540     * </pre>
541     * @return a Context associated with the current thread
542     * @see Context#getCurrentContext()
543     * @see Context#exit()
544     * @see #call(ContextAction)
545     */
546    public Context enterContext()
547    {
548        return enterContext(null);
549    }
