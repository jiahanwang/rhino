444                                   BitSet useBeforeDef, BitSet notDef) {
445        int card = entrySet.cardinality();
446        entrySet.or(exitSet);
447        entrySet.and(notDef);
448        entrySet.or(useBeforeDef);
449        return entrySet.cardinality() != card;
450    }

452    /*
453        the type of an expression is relatively unknown. Cases we can be sure
454        about are -
455            Literals,
456            Arithmetic operations - always return a Number
457    */
458    private static int findExpressionType(OptFunctionNode fn, Node n,
459                                          int[] varTypes)
460    {
461        switch (n.getType()) {
462            case Token.NUMBER:
463                return Optimizer.NumberType;
464 
465            case Token.CALL:
466            case Token.NEW:
467            case Token.REF_CALL:
468                return Optimizer.AnyType;
469 
470            case Token.GETELEM:
471            case Token.GETPROP:
472            case Token.NAME:
473            case Token.THIS:
474                return Optimizer.AnyType;
475 
476            case Token.GETVAR:
477                return varTypes[fn.getVarIndex(n)];
478 
479            case Token.INC:
480            case Token.DEC:
481            case Token.MUL:
482            case Token.DIV:
483            case Token.MOD:
484            case Token.BITOR:
485            case Token.BITXOR:
486            case Token.BITAND:
487            case Token.BITNOT:
488            case Token.LSH:
489            case Token.RSH:
490            case Token.URSH:
491            case Token.SUB:
492            case Token.POS:
493            case Token.NEG:
494                return Optimizer.NumberType;
495 
496            case Token.VOID:
497                // NYI: undefined type
498                return Optimizer.AnyType;
499 
500            case Token.FALSE:
501            case Token.TRUE:
502            case Token.EQ:
503            case Token.NE:
504            case Token.LT:
505            case Token.LE:
506            case Token.GT:
507            case Token.GE:
508            case Token.SHEQ:
509            case Token.SHNE:
510            case Token.NOT:
511            case Token.INSTANCEOF:
512            case Token.IN:
513            case Token.DEL_REF:
514            case Token.DELPROP:
515                // NYI: boolean type
516                return Optimizer.AnyType;
517 
518            case Token.STRING:
519            case Token.TYPEOF:
520            case Token.TYPEOFNAME:
521                // NYI: string type
522                return Optimizer.AnyType;
523 
524            case Token.NULL:
525            case Token.REGEXP:
526            case Token.ARRAYCOMP:
527            case Token.ARRAYLIT:
528            case Token.OBJECTLIT:
529                return Optimizer.AnyType; // XXX: actually, we know it's not
530            // number, but no type yet for that
531 
532            case Token.ADD: {
533                // if the lhs & rhs are known to be numbers, we can be sure that's
534                // the result, otherwise it could be a string.
535                Node child = n.getFirstChild();
536                int lType = findExpressionType(fn, child, varTypes);
537                int rType = findExpressionType(fn, child.getNext(), varTypes);
538                return lType | rType;    // we're not distinguishing strings yet
539            }
540 
541            case Token.HOOK: {
542                Node ifTrue = n.getFirstChild().getNext();
543                Node ifFalse = ifTrue.getNext();
544                int ifTrueType = findExpressionType(fn, ifTrue, varTypes);
545                int ifFalseType = findExpressionType(fn, ifFalse, varTypes);
546                return ifTrueType | ifFalseType;
547            }
548 
549            case Token.COMMA:
550            case Token.SETVAR:
551            case Token.SETNAME:
552            case Token.SETPROP:
553            case Token.SETELEM:
554                return findExpressionType(fn, n.getLastChild(), varTypes);
555 
556            case Token.AND:
557            case Token.OR: {
558                Node child = n.getFirstChild();
559                int lType = findExpressionType(fn, child, varTypes);
560                int rType = findExpressionType(fn, child.getNext(), varTypes);
561                return lType | rType;
562            }
563        }
564 
565        return Optimizer.AnyType;
566    }

568    private static boolean findDefPoints(OptFunctionNode fn, Node n,
569                                         int[] varTypes)
570    {
571        boolean result = false;
572        Node first = n.getFirstChild();
573        for (Node next = first; next != null; next = next.getNext()) {
574            result |= findDefPoints(fn, next, varTypes);
575        }
576        switch (n.getType()) {
577            case Token.DEC :
578            case Token.INC :
579                if (first.getType() == Token.GETVAR) {
580                    // theVar is a Number now
581                    int i = fn.getVarIndex(first);
582                    result |= assignType(varTypes, i, Optimizer.NumberType);
583                }
584                break;
585            case Token.SETVAR : {
586                Node rValue = first.getNext();
587                int theType = findExpressionType(fn, rValue, varTypes);
588                int i = fn.getVarIndex(n);
589                result |= assignType(varTypes, i, theType);
590                break;
591            }
592        }
593        return result;
594    }

596    private boolean doTypeFlow(OptFunctionNode fn, Node[] statementNodes,
597                               int[] varTypes)
598    {
599        boolean changed = false;
600 
601        for (int i = itsStartNodeIndex; i <= itsEndNodeIndex; i++) {
602            Node n = statementNodes[i];
603            if (n != null) {
604                changed |= findDefPoints(fn, n, varTypes);
605            }
606        }
607 
608        return changed;
609    }
